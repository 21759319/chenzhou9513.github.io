MPI Hello world

简介

MPI 安装 和 hello world

安装

需要GCC和Homebrew

brew install openmpi

hello_world.c

#include <mpi.h>
#include <stdio.h>

int main(int argc, char** argv) {

// Initialize the MPI environment
MPI_Init(NULL, NULL);

// Get the number of processes
int world_size;
MPI_Comm_size(MPI_COMM_WORLD, &world_size);

// Get the rank of the process
int world_rank;
MPI_Comm_rank(MPI_COMM_WORLD, &world_rank);

// Get the name of the processor
char processor_name[MPI_MAX_PROCESSOR_NAME];
int name_len;
MPI_Get_processor_name(processor_name, &name_len);

// Print off a hello world message
printf("Hello world from processor %s, rank %d out of %d processors\n",
       processor_name, world_rank, world_size);

// Finalize the MPI environment.
MPI_Finalize();
}

引入头文件

#include <mpi.h>

初始化MPI全局环境,初始化进程,给进程分配ranks,传递mpi运行参数

MPI_Init(
    int* argc,
    char*** argv)

返回communicator的大小到size,MPI_Comm是MPI进程的通信子,MPI_COMM_WORLD表示有所进程通信子的集合

MPI_Comm_size(
    MPI_Comm communicator,
    int* size)

MPI_Comm_rank获取通信子所绑定进程的rank,rank从0开始，主要用于识别进程

MPI_Comm_rank(
    MPI_Comm communicator,
    int* rank)

MPI_Get_processor_name获取当前正在运行的进程的name,不常用

MPI_Get_processor_name(
    char* name,
    int* name_length)

MPI_Finalize 释放MPI环境

MPI_Finalize()

运行

makefile

EXECS=hello_world
MPICC?=mpicc

all: ${EXECS}

hello_world: hello_world.c
    ${MPICC} -o hello_world hello_world.c

clean:
    rm ${EXECS}

简单编译运行

mpicc hello_world.c -o helloword
mpirun -n 2 ./helloworld

如果是cpp

mpic++ hello_world.c -o helloword
mpirun -n 2 ./helloworld
